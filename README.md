-- SISTEMA DE CARTOES - ORACLE

-- Criar tabelas (sem DROP DATABASE, apenas DROP TABLE se quiser resetar)
DROP TABLE transacao CASCADE CONSTRAINTS;
DROP TABLE cartao CASCADE CONSTRAINTS;
DROP TABLE cliente CASCADE CONSTRAINTS;

CREATE TABLE cliente (
  id    NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  cpf   VARCHAR2(11)  UNIQUE NOT NULL,
  nome  VARCHAR2(120) NOT NULL,
  email VARCHAR2(160) NOT NULL,
  renda NUMBER(10,2)  NOT NULL
);

CREATE TABLE cartao (
  id                NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  numero            VARCHAR2(20)  UNIQUE NOT NULL,
  bandeira          VARCHAR2(20)  NOT NULL,
  limite_total      NUMBER(10,2) NOT NULL,
  limite_disponivel NUMBER(10,2) NOT NULL,
  cliente_id        NUMBER NOT NULL,
  CONSTRAINT fk_cartao_cliente FOREIGN KEY (cliente_id) REFERENCES cliente(id)
);

CREATE TABLE transacao (
  id        NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  cartao_id NUMBER NOT NULL,
  data      TIMESTAMP NOT NULL,
  descricao VARCHAR2(255) NOT NULL,
  valor     NUMBER(10,2) NOT NULL,
  status    VARCHAR2(20) NOT NULL,
  CONSTRAINT fk_transacao_cartao FOREIGN KEY (cartao_id) REFERENCES cartao(id)
);
